# Tensorflow web server with Flask


# Installation


    sudo sh requiriments.sh

    
# Manual installation


    sudo apt-get install python3
    sudo apt-get install python3-pip

    sudo pip3 install 

# Run Web Server

    sh run.sh 

# Open on browser

    xdg-open http://localhost:5000/
